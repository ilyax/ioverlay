ioverlay
========

jquery tiny lightbox

  
    $(document).ready(function ()   
        {
           $('#btnOpenObj').click(function (e) {
                $('#yourDemoDiv').ioverlay();
            });

            $('#btnCloseObj').click(function (e) {
                $('#yourDemoDiv').ioverlay('close');
            });
        }
    );
